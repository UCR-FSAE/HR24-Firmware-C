256#define SHAMD5_MODE_CLOSE_HASH 0x00000010 // Performs the padding, the

258 // the end of the block, as per
259 // MD5/SHA-1/SHA-2 specification
260#define SHAMD5_MODE_ALGO_CONSTANT \
261 0x00000008 // The initial digest register will
262 // be overwritten with the

264 // selected algorithm when hashing
265 // and the initial digest count
266 // register will be reset to 0
267#define SHAMD5_MODE_ALGO_M 0x00000007 // Hash Algorithm

271#define SHAMD5_MODE_ALGO_SHA256 0x00000006 // SHA-256
